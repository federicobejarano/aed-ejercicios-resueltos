/*

Se dispone de una secuencia de caracteres.
Se desea determinar la cantidad de palabras que comienzan con la letra 'I'.

*/


ACCION 2.1.9 ES

	AMBIENTE

		sec : secuencia de caracter
		v, ant : caracter
		c : entero

	PROCESO

		ARR(sec)
		AVZ(sec, v)
		c := 0

		Mientras NFDS (sec) hacer
			Si v = "I" Entonces
				c := c + 1
			FinSi

			Mientras v <> " " hacer
				AVZ(sec, v)
			FinMientras

			AVZ(sec, v)
		FinMientras

		ESC("La secuencia contiene ", c, " palabras que empiezan con I")
		Cerrar(sec)

FinAccion
